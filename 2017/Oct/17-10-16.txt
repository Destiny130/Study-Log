                  $(obj).children("#showDetails").css("display", "none");
              }
          </script>
          <td id=@item.TraceKey onmouseover="getDetails(this)" onmouseout="mouseleft(this)">

              @if (item.RequestMsg.Trim() != null && item.RequestMsg.Trim() != "")

                  </p>
              }
          </td>
